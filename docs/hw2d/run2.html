 74    continue_file: bool = False,

 92        "kinetic_energy",
 93        "enstrophy",
 94        "enstrophy_phi",

 96    # Plotting
 97    t0_std = 800,
 98    plot_properties: Iterable[str] = (

102        "enstrophy",
103        "energy",
104        "kinetic_energy",
105        "thermal_energy",
106    ),
107    # Other
108    debug: bool = False,
109    force_recompute: bool = False,
110):
111    """
112    Run the simulation with the given parameters.
113
114    Args:
115        step_size (float, optional): Incremental step for simulation progression. Defaults to 0.025.
116        end_time (float, optional): Duration till the simulation should run. Defaults to 1_000.
117        grid_pts (int, optional): Grid points. Suggested: 128 for coarse, 1024 for fine. Defaults to 512.
118        k0 (float, optional): Determines k-focus. Suggested: 0.15 for high-k, 0.0375 for low-k. Defaults to 0.15.
119        N (int, optional): Dissipation exponent's half value. Defaults to 3.
120        nu (float, optional): Viscosity. Suggested: 5e-10 for coarse-large, 1e-4 for fine-small. Defaults to 5.0e-08.
121        c1 (float, optional): Transition scale between hydrodynamic and adiabatic. Suggested values: 0.1, 1, 5. Defaults to 1.0.
122        kappa_coeff (float, optional): Coefficient of d/dy phi. Defaults to 1.0.
123        poisson_bracket_coeff (float, optional): Coefficient of Poisson bracket [A,B] implemented with Arakawa Scheme. Defaults to 1.0.
124        zonal (bool, ooptional): If True, uses make_zonal_func to simulate mHW2D. Defaults to False.
125        seed (int or None, optional): Seed for random number generation. Defaults to None.
126        init_type (str, optional): Initialization method. Choices: 'fourier', 'sine', 'random', 'normal'. Defaults to 'normal'.
127        init_scale (float, optional): Scaling factor for initialization. Defaults to 0.01.
128        output_path (str, optional): Where to save the simulation data. Defaults to ''.
129        recording_start_time (float, optional): Time (t) from which onwards the data is recorded. Defaults to 0.
130        continue_file (bool, optional): If True, continue with existing file. Defaults to False.
131        buffer_length (int, optional): Size of buffer for storage. Defaults to 100.
132        chunks (int, optional): Chunks of the h5 file. Defaults to 100.
133        downsample_factor (float, optional): Factor along each axis that it is downsampled with for saving. Defaults to 1.
134        add_last_state (bool, optional): Whether the last high-resolution frame is saved. Turns True if downsampling on. Defaults to False.
135        snaps (int, optional): Snapshot intervals for saving. Defaults to 1.
136        movie (bool, optional): If True, generate a movie out of simulation. Defaults to True.
137        min_fps (int, optional): Parameter for movie generation. Defaults to 10.
138        dpi (int, optional): Parameter for movie generation. Defaults to 75.
139        speed (int, optional): Parameter for movie generation. Defaults to 5.
140        properties (Iterable[str], optional): List of properties to calculate for the saved file.
141        plot_properties (Iterable[str], optional): List of properties to plot a timetrace for.
142        debug (bool, optional): Enable or disable debug mode. Defaults to False.
143
144    Returns:
145        None: The function saves simulation data or generates a movie as specified.
146    """
147    step_size = float(step_size)
148    nu = float(nu)
149
150    # Unpacking
151    y = grid_pts
152    x = grid_pts
153    L = 2 * np.pi / k0  # Physical box size
154    dx = L / x  # Grid resolution
155    steps = int(end_time / step_size)  # Number of Steps until end_time
156    snap_count = steps // snaps + 1  # number of snapshots
157    field_list = ("density", "omega", "phi")
158    initial_time = np.float64(0)
159    iteration_count = 0
160    np.random.seed(seed)
161    if downsample_factor != 1:
162        add_last_state = True
163    if show_property not in properties:
164        properties += [show_property]
165
166    # Define Initializations
167    noise = {
168        "fourier": lambda y, x: get_fft_noise(
169            resolution=[y, x],
170            size=L,
171            scale=1,
172            min_wavelength=dx * 12,
173            max_wavelength=dx * grid_pts * 100,
174            factor=2,
175        ),
176        "sine": lambda y, x: get_2d_sine((y, x), L),
177        "random": lambda y, x: np.random.rand(y, x).astype(np.float64),
178        "normal": lambda y, x: np.random.normal(size=(y, x)).astype(np.float64),
179    }
180    downsample_fnc = fourier_downsample
181
182    # Physics
183    # TODO: Move all to Tensors along batch dimension for simulations!
184    physics_params = dict(
185        dx=dx,
186        N=N,
187        c1=c1,
188        nu=nu,
189        k0=k0,
190        poisson_bracket_coeff=poisson_bracket_coeff,
191        kappa_coeff=kappa_coeff,
192        zonal=zonal
193    )
194    # Initialize Plasma
195    plasma = Namespace(
196        density=noise[init_type](y, x) * init_scale,
197        omega=noise[init_type](y, x) * init_scale,
198        phi=noise[init_type](y, x) * init_scale,
199        age=0,
200    )
201
202    # File Handling
203    if output_path:
204        y_save = y
205        x_save = x
206        if downsample_factor != 1:
207            y_save = int(round(y / downsample_factor))
208            x_save = int(round(x / downsample_factor))
209            print(f"Downsample by {downsample_factor}x: {y}x{x}->{y_save}x{x_save}")
210        # Output data from this run
211        buffer = {
212            field: np.zeros((buffer_length, y_save, x_save), dtype=np.float32)
213            for field in field_list
214        }
215        # Property buffer
216        buffer[f"time"] = np.zeros((buffer_length, 1), dtype=np.float32)
217        for prop_name in properties:
218            buffer[f"fullres_{prop_name}"] = np.zeros((buffer_length, 1), dtype=np.float32)
219        # Other output parameters
220        output_params = {
221            "buffer": buffer,
222            "buffer_index": 0,
223            "output_path": output_path,
224        }
225        # Load Data
226        if os.path.isfile(output_path) and not force_recompute:
227            if continue_file:
228                plasma, physics_params = continue_h5_file(output_path, field_list)
229                save_params = get_save_params(
230                    physics_params, step_size, snaps, x, y, x_save=x_save, y_save=y_save, recording_start_time=recording_start_time, adaptive_step_size=adaptive_step_size,
231                )
232                initial_time = np.float64(plasma.age)
233                print(
234                    f"Loaded: {output_path} (age={plasma.age}, time={initial_time})\n{physics_params}"
235                )
236                # Check if broken
237                for field in plasma.keys():
238                    # Not NaNs
239                    if np.isnan(np.sum(plasma[field])):
240                        print(f"Input file is broken: NaNs in {field}")
241                        raise BaseException(f"Input file is broken: NaNs in {field}")
242                    # Not zeros
243                    if field in ("density", "omega"):
244                        if np.all(plasma[field] == 0):
245                            print(f"FAILED {field} is zero")
246                            raise BaseException(f"Input File is zero: {field}")
247            else:
248                print(f"File already exists.")
249                return
250        # Create Data
251        else:
252            save_params = get_save_params(
253                physics_params, step_size, snaps, x, y, x_save=x_save, y_save=y_save, recording_start_time=recording_start_time, adaptive_step_size=adaptive_step_size,            )
254            # Create file
255            create_appendable_h5(
256                output_path,
257                save_params,
258                properties=["time"] + [f"fullres_{prop_name}" for prop_name in properties],
259                dtype=np.float32,
260                chunk_size=chunk_size,
261                add_last_state=add_last_state,
262            )
263            # Save initial values
264            if not recording_start_time:
265                new_val = Namespace(
266                    **{k: v for k, v in plasma.items() if k in ("phi", "omega", "density")}
267                )
268                last_state = {}
269                for k, v in plasma.items():
270                    if k in ("phi", "omega", "density"):
271                        if downsample_factor != 1:
272                            new_val[k] = downsample_fnc(v, downsample_factor)
273                        if add_last_state:
274                            last_state[f"state_{k}"] = v
275                for prop_name in properties:
276                    new_val[f"fullres_{prop_name}"] = property_fncs[prop_name](
277                        n=plasma["density"],
278                        p=plasma["phi"],
279                        o=plasma["omega"],
280                        dx=dx,
281                        c1=c1,
282                    )
283                new_val["time"] = initial_time
284                output_params["buffer_index"] = save_to_buffered_h5(
285                    new_val=new_val,
286                    last_state=last_state,
287                    buffer_length=buffer_length,
288                    **output_params,
289                )
290
291    # Display runtime parameters
292    # save_params["output_path"] = output_path  # No support for strings
293    format_print_dict(save_params)
294
295    # Setup Simulation
296    hw = HW(**physics_params, debug=debug)
297    plasma["phi"] = hw.get_phi(plasma.omega, physics_params["dx"])
298
299    # Setup Progress Bar
300    print("Running simulation...")
301    iteration_count = 0
302    bar_format = "{rate_fmt} | {desc} {percentage:>6.2f}%|{bar}| {n:.2f}/{total:.2f} [{elapsed}<{remaining}]"
303    # bar_format = "{percentage:>6.2f}%|{bar}| {n:.4f}/{total_fmt} [{elapsed}<{remaining}] ({rate_fmt})"
304    used_step_size = step_size
305    used_time_steps = []
306    current_time = initial_time
307    new_val = {}
308
309    try:
310        with tqdm(
311            total=end_time - initial_time,
312            unit="t",
313            bar_format=bar_format,
314        ) as pbar:
315            # Run Simulation
316            while round(current_time, 4) < end_time:
317                # Progress one step, alternatively: hw.euler_step()
318                successful = False
319                while not successful:
320                    try:
321                        plasma = hw.rk4_step(plasma, dt=step_size, dx=dx)
322                        used_step_size = step_size
323                        successful = True
324                    except Exception as e:
325                        print(e)
326                        time.sleep(10)
327                iteration_count += 1
328
329                # Improved accuracy of time tracking
330                if adaptive_step_size:
331                    used_time_steps.append(np.float64(used_step_size))
332                    current_time = initial_time + np.sum(used_time_steps)
333                else:
334                    current_time = initial_time + iteration_count * step_size
335
336                # Batched Processing
337                # TODO: Figure out whether it should be iteration_count % (snaps - 1) == 0
338                if (current_time >= recording_start_time) and (iteration_count % snaps == 0):
339                    new_val = Namespace(
340                        **{
341                            k: v
342                            for k, v in plasma.items()
343                            if k in ("phi", "omega", "density")
344                        }
345                    )
346                    # Add properties if they were selected
347                    # TODO: Faster to do before saving through trivial parallelization
348                    new_val["time"] = current_time
349                    for prop_name in properties:
350                        new_val[f"fullres_{prop_name}"] = property_fncs[prop_name](
351                            n=plasma["density"],
352                            p=plasma["phi"],
353                            o=plasma["omega"],
354                            dx=dx,
355                            c1=c1,
356                        )
357
358                    # Save to records
359                    if output_path:
360                        last_state = {}
361                        # Save downsampled & state
362                        for k, v in plasma.items():
363                            if k in ("phi", "omega", "density"):
364                                if downsample_factor != 1:
365                                    new_val[k] = downsample_fnc(v, downsample_factor)
366                                if add_last_state:
367                                    last_state[f"state_{k}"] = v
368                        output_params["buffer_index"] = save_to_buffered_h5(
369                            new_val=new_val,
370                            last_state=last_state,
371                            buffer_length=buffer_length,
372                            **output_params,
373                        )
374
375                    # Check for breaking
376                    if np.isnan(np.sum(plasma.density)):
377                        print(f"FAILED @ {iteration_count:,} steps ({plasma.age:,})")
378                        break
379
380                # Calculate iterations per second, handling division by zero
381                try:
382                    iter_per_sec = iteration_count / pbar.format_dict["elapsed"]
383                except ZeroDivisionError:
384                    iter_per_sec = float("inf")  # or set to 0 or any default value
385
386                # Update progress
387                pbar.update(used_step_size)
388                new_description = f"{iter_per_sec:.2f}it/s"
389                if show_property:
390                    new_description += f" | Γn = {new_val.get(show_property, 0):.2g}"
391                pbar.set_description(new_description)
392
393    except Exception as e:
394        print(e)
395        raise e
396        # os.remove(output_path)
397        # print(f"removed {output_path}")
398        return
399
400    # If output_path is defined, flush any remaining data in the buffer
401    if output_path and (current_time > recording_start_time) and (output_params["buffer_index"] > 0):
402        append_h5(**output_params)
403
404    # Get Performance stats
405    hw.print_log()
406
407    # Generate Movie from saved file
408    if movie and output_path:
409        print(f"Generating movie...")
410        create_movie(
411            input_filename=output_path,
412            output_filename=output_path.replace(".h5", ""),
413            t0=0,
414            t1=None,
415            plot_order=field_list,
416            min_fps=min_fps,
417            dpi=dpi,
418            speed=speed,
419        )
420
421    if properties and output_path:
422        print(f"Calculating properties...")
423        calculate_properties(
424            file_path=output_path,
425            batch_size=buffer_length,
426            property_list=properties,
427            force_recompute=True,
428            is_debug=False,
429        )
430
431    if plot_properties and output_path:
432        print(f"Plotting properties...")
433        plot_timetrace_comparison(
434            file_path=output_path,
435            out_path=None,
436            properties=plot_properties,
437            t0=0,
438            t0_std=t0_std,
439        )
440

443    fire.Fire(run)

156    steps = int(end_time / step_size)  # Number of Steps until end_time
157    snap_count = steps // snaps + 1  # number of snapshots
158    field_list = ("density", "omega", "phi")
159    initial_time = np.float64(0)
160    iteration_count = 0
161    np.random.seed(seed)
162    if downsample_factor != 1:
163        add_last_state = True
164    if show_property not in properties:
165        properties += [show_property]
166
167    # Define Initializations
168    noise = {
169        "fourier": lambda y, x: get_fft_noise(
170            resolution=[y, x],
171            size=L,
172            scale=1,
173            min_wavelength=dx * 12,
174            max_wavelength=dx * grid_pts * 100,
175            factor=2,
176        ),
177        "sine": lambda y, x: get_2d_sine((y, x), L),
178        "random": lambda y, x: np.random.rand(y, x).astype(np.float64),
179        "normal": lambda y, x: np.random.normal(size=(y, x)).astype(np.float64),
180    }
181    downsample_fnc = fourier_downsample
182
183    # Physics
184    # TODO: Move all to Tensors along batch dimension for simulations!
185    physics_params = dict(
186        dx=dx,
187        N=N,
188        c1=c1,
189        nu=nu,
190        k0=k0,
191        poisson_bracket_coeff=poisson_bracket_coeff,
192        kappa_coeff=kappa_coeff,
193        zonal=zonal
194    )
195    # Initialize Plasma
196    plasma = Namespace(
197        density=noise[init_type](y, x) * init_scale,
198        omega=noise[init_type](y, x) * init_scale,
199        phi=noise[init_type](y, x) * init_scale,
200        age=0,
201    )
202
203    # File Handling
204    if output_path:
205        y_save = y
206        x_save = x
207        if downsample_factor != 1:
208            y_save = int(round(y / downsample_factor))
209            x_save = int(round(x / downsample_factor))
210            print(f"Downsample by {downsample_factor}x: {y}x{x}->{y_save}x{x_save}")
211        # Output data from this run
212        buffer = {
213            field: np.zeros((buffer_length, y_save, x_save), dtype=np.float32)
214            for field in field_list
215        }
216        # Property buffer
217        buffer[f"time"] = np.zeros((buffer_length, 1), dtype=np.float32)
218        for prop_name in properties:
219            buffer[f"fullres_{prop_name}"] = np.zeros((buffer_length, 1), dtype=np.float32)
220        # Other output parameters
221        output_params = {
222            "buffer": buffer,
223            "buffer_index": 0,
224            "output_path": output_path,
225        }
226        # Load Data
227        if os.path.isfile(output_path) and not force_recompute:
228            if continue_file:
229                plasma, physics_params = continue_h5_file(output_path, field_list)
230                save_params = get_save_params(
231                    physics_params, step_size, snaps, x, y, x_save=x_save, y_save=y_save, recording_start_time=recording_start_time, adaptive_step_size=adaptive_step_size,
232                )
233                initial_time = np.float64(plasma.age)
234                print(
235                    f"Loaded: {output_path} (age={plasma.age}, time={initial_time})\n{physics_params}"
236                )
237                # Check if broken
238                for field in plasma.keys():
239                    # Not NaNs
240                    if np.isnan(np.sum(plasma[field])):
241                        print(f"Input file is broken: NaNs in {field}")
242                        raise BaseException(f"Input file is broken: NaNs in {field}")
243                    # Not zeros
244                    if field in ("density", "omega"):
245                        if np.all(plasma[field] == 0):
246                            print(f"FAILED {field} is zero")
247                            raise BaseException(f"Input File is zero: {field}")
248            else:
249                print(f"File already exists.")
250                return
251        # Create Data
252        else:
253            save_params = get_save_params(
254                physics_params, step_size, snaps, x, y, x_save=x_save, y_save=y_save, recording_start_time=recording_start_time, adaptive_step_size=adaptive_step_size,            )
255            # Create file
256            create_appendable_h5(
257                output_path,
258                save_params,
259                properties=["time"] + [f"fullres_{prop_name}" for prop_name in properties],
260                dtype=np.float32,
261                chunk_size=chunk_size,
262                add_last_state=add_last_state,
263            )
264            # Save initial values
265            if not recording_start_time:
266                new_val = Namespace(
267                    **{k: v for k, v in plasma.items() if k in ("phi", "omega", "density")}
268                )
269                last_state = {}
270                for k, v in plasma.items():
271                    if k in ("phi", "omega", "density"):
272                        if downsample_factor != 1:
273                            new_val[k] = downsample_fnc(v, downsample_factor)
274                        if add_last_state:
275                            last_state[f"state_{k}"] = v
276                for prop_name in properties:
277                    new_val[f"fullres_{prop_name}"] = property_fncs[prop_name](
278                        n=plasma["density"],
279                        p=plasma["phi"],
280                        o=plasma["omega"],
281                        dx=dx,
282                        c1=c1,
283                    )
284                new_val["time"] = initial_time
285                output_params["buffer_index"] = save_to_buffered_h5(
286                    new_val=new_val,
287                    last_state=last_state,
288                    buffer_length=buffer_length,
289                    **output_params,
290                )
291
292    # Display runtime parameters
293    # save_params["output_path"] = output_path  # No support for strings
294    format_print_dict(save_params)
295
296    # Setup Simulation
297    hw = HW(**physics_params, debug=debug)
298    plasma["phi"] = hw.get_phi(plasma.omega, physics_params["dx"])
299
300    # Setup Progress Bar
301    print("Running simulation...")
302    iteration_count = 0
303    bar_format = "{rate_fmt} | {desc} {percentage:>6.2f}%|{bar}| {n:.2f}/{total:.2f} [{elapsed}<{remaining}]"
304    # bar_format = "{percentage:>6.2f}%|{bar}| {n:.4f}/{total_fmt} [{elapsed}<{remaining}] ({rate_fmt})"
305    used_step_size = step_size
306    used_time_steps = []
307    current_time = initial_time
308    new_val = {}
309
310    try:
311        with tqdm(
312            total=end_time - initial_time,
313            unit="t",
314            bar_format=bar_format,
315        ) as pbar:
316            # Run Simulation
317            while round(current_time, 4) < end_time:
318                # Progress one step, alternatively: hw.euler_step()
319                successful = False
320                while not successful:
321                    try:
322                        plasma = hw.rk4_step(plasma, dt=step_size, dx=dx)
323                        used_step_size = step_size
324                        successful = True
325                    except Exception as e:
326                        print(e)
327                        time.sleep(10)
328                iteration_count += 1
329
330                # Improved accuracy of time tracking
331                if adaptive_step_size:
332                    used_time_steps.append(np.float64(used_step_size))
333                    current_time = initial_time + np.sum(used_time_steps)
334                else:
335                    current_time = initial_time + iteration_count * step_size
336
337                # Batched Processing
338                # TODO: Figure out whether it should be iteration_count % (snaps - 1) == 0
339                if (current_time >= recording_start_time) and (iteration_count % snaps == 0):
340                    new_val = Namespace(
341                        **{
342                            k: v
343                            for k, v in plasma.items()
344                            if k in ("phi", "omega", "density")
345                        }
346                    )
347                    # Add properties if they were selected
348                    # TODO: Faster to do before saving through trivial parallelization
349                    new_val["time"] = current_time
350                    for prop_name in properties:
351                        new_val[f"fullres_{prop_name}"] = property_fncs[prop_name](
352                            n=plasma["density"],
353                            p=plasma["phi"],
354                            o=plasma["omega"],
355                            dx=dx,
356                            c1=c1,
357                        )
358
359                    # Save to records
360                    if output_path:
361                        last_state = {}
362                        # Save downsampled & state
363                        for k, v in plasma.items():
364                            if k in ("phi", "omega", "density"):
365                                if downsample_factor != 1:
366                                    new_val[k] = downsample_fnc(v, downsample_factor)
367                                if add_last_state:
368                                    last_state[f"state_{k}"] = v
369                        output_params["buffer_index"] = save_to_buffered_h5(
370                            new_val=new_val,
371                            last_state=last_state,
372                            buffer_length=buffer_length,
373                            **output_params,
374                        )
375
376                    # Check for breaking
377                    if np.isnan(np.sum(plasma.density)):
378                        print(f"FAILED @ {iteration_count:,} steps ({plasma.age:,})")
379                        break
380
381                # Calculate iterations per second, handling division by zero
382                try:
383                    iter_per_sec = iteration_count / pbar.format_dict["elapsed"]
384                except ZeroDivisionError:
385                    iter_per_sec = float("inf")  # or set to 0 or any default value
386
387                # Update progress
388                pbar.update(used_step_size)
389                new_description = f"{iter_per_sec:.2f}it/s"
390                if show_property:
391                    new_description += f" | Γn = {new_val.get(show_property, 0):.2g}"
392                pbar.set_description(new_description)
393
394    except Exception as e:
395        print(e)
396        raise e
397        # os.remove(output_path)
398        # print(f"removed {output_path}")
399        return
400
401    # If output_path is defined, flush any remaining data in the buffer
402    if output_path and (current_time > recording_start_time) and (output_params["buffer_index"] > 0):
403        append_h5(**output_params)
404
405    # Get Performance stats
406    hw.print_log()
407
408    # Generate Movie from saved file
409    if movie and output_path:
410        print(f"Generating movie...")
411        create_movie(
412            input_filename=output_path,
413            output_filename=output_path.replace(".h5", ""),
414            t0=0,
415            t1=None,
416            plot_order=field_list,
417            min_fps=min_fps,
418            dpi=dpi,
419            speed=speed,
420        )
421
422    if properties and output_path:
423        print(f"Calculating properties...")
424        calculate_properties(
425            file_path=output_path,
426            batch_size=buffer_length,
427            property_list=properties,
428            force_recompute=True,
429            is_debug=False,
430        )
431
432    if plot_properties and output_path:
433        print(f"Plotting properties...")
434        plot_timetrace_comparison(
435            file_path=output_path,
436            out_path=None,
437            properties=plot_properties,
438            t0=0,
439            t0_std=t0_std,
440        )

Run the simulation with the given parameters.

Arguments:
  • step_size (float, optional): Incremental step for simulation progression. Defaults to 0.025.
  • end_time (float, optional): Duration till the simulation should run. Defaults to 1_000.
  • grid_pts (int, optional): Grid points. Suggested: 128 for coarse, 1024 for fine. Defaults to 512.
  • k0 (float, optional): Determines k-focus. Suggested: 0.15 for high-k, 0.0375 for low-k. Defaults to 0.15.
  • N (int, optional): Dissipation exponent's half value. Defaults to 3.
  • nu (float, optional): Viscosity. Suggested: 5e-10 for coarse-large, 1e-4 for fine-small. Defaults to 5.0e-08.
  • c1 (float, optional): Transition scale between hydrodynamic and adiabatic. Suggested values: 0.1, 1, 5. Defaults to 1.0.
  • kappa_coeff (float, optional): Coefficient of d/dy phi. Defaults to 1.0.
  • poisson_bracket_coeff (float, optional): Coefficient of Poisson bracket [A,B] implemented with Arakawa Scheme. Defaults to 1.0.
  • zonal (bool, ooptional): If True, uses make_zonal_func to simulate mHW2D. Defaults to False.
  • seed (int or None, optional): Seed for random number generation. Defaults to None.
  • init_type (str, optional): Initialization method. Choices: 'fourier', 'sine', 'random', 'normal'. Defaults to 'normal'.
  • init_scale (float, optional): Scaling factor for initialization. Defaults to 0.01.
  • output_path (str, optional): Where to save the simulation data. Defaults to ''.
  • recording_start_time (float, optional): Time (t) from which onwards the data is recorded. Defaults to 0.
  • continue_file (bool, optional): If True, continue with existing file. Defaults to False.
  • buffer_length (int, optional): Size of buffer for storage. Defaults to 100.
  • chunks (int, optional): Chunks of the h5 file. Defaults to 100.
  • downsample_factor (float, optional): Factor along each axis that it is downsampled with for saving. Defaults to 1.
  • add_last_state (bool, optional): Whether the last high-resolution frame is saved. Turns True if downsampling on. Defaults to False.
  • snaps (int, optional): Snapshot intervals for saving. Defaults to 1.
  • movie (bool, optional): If True, generate a movie out of simulation. Defaults to True.
  • min_fps (int, optional): Parameter for movie generation. Defaults to 10.
  • dpi (int, optional): Parameter for movie generation. Defaults to 75.
  • speed (int, optional): Parameter for movie generation. Defaults to 5.
  • properties (Iterable[str], optional): List of properties to calculate for the saved file.
  • plot_properties (Iterable[str], optional): List of properties to plot a timetrace for.
  • debug (bool, optional): Enable or disable debug mode. Defaults to False.
Returns:

None: The function saves simulation data or generates a movie as specified.
